Hash for Immutability — PowerShell

$ts = Get-Date -Format "yyyyMMdd-HHmmss"
$dir = "$env:TEMP/xrbc-ai-$ts"; New-Item -ItemType Directory -Path $dir | Out-Null
$files = @(
  "https://xrbitcoincash.com/ai/ai/health.html",
  "https://xrbitcoincash.com/ai/ai/ai/price.html"
)
$hashes = @()
foreach ($u in $files) {
  $name = (Split-Path $u -Leaf)
  $dst = Join-Path $dir $name
  Invoke-WebRequest -UseBasicParsing $u -OutFile $dst
  $h = Get-FileHash -Path $dst -Algorithm SHA256
  $hashes += [PSCustomObject]@{ file=$name; sha256=$h.Hash }
}
$hashes | Format-Table -AutoSize
$hashes | ConvertTo-Json | Out-File (Join-Path $dir "checksums.json") -Encoding utf8
Write-Host "Checksums saved to: $dir\checksums.json"

Hash for Immutability — Bash

TS=$(date +%Y%m%d-%H%M%S)
DIR="/tmp/xrbc-ai-$TS"; mkdir -p "$DIR"
FILES=(
  "https://xrbitcoincash.com/ai/ai/health.html"
  "https://xrbitcoincash.com/ai/ai/ai/price.html"
)
for U in "${FILES[@]}"; do
  NAME="$(basename "$U")";
  curl -sL "$U" -o "$DIR/$NAME"
  printf "%s  %s\n" "$(sha256sum "$DIR/$NAME" | cut -d' ' -f1)" "$NAME"
done | tee "$DIR/checksums.txt"
echo "Checksums saved to: $DIR/checksums.txt"

5) Local One‑Shot Audit Helpers

PowerShell Function

function Test-XRBC {
  $base = "https://xrbitcoincash-github-io.onrender.com"
  $siteHealth = "https://xrbitcoincash.com/ai/ai/health.html"
  $sitePrice  = "https://xrbitcoincash.com/ai/ai/ai/price.html"
  $r1 = Invoke-WebRequest -UseBasicParsing "$base/healthz" -TimeoutSec 15
  $r2 = Invoke-WebRequest -UseBasicParsing $siteHealth -TimeoutSec 15
  $r3 = Invoke-WebRequest -UseBasicParsing $sitePrice -TimeoutSec 15
  [PSCustomObject]@{ proxy_health=$r1.StatusCode; ai_health=$r2.StatusCode; ai_price=$r3.StatusCode }
}
# Usage
Test-XRBC | Format-List

Bash One‑Liner
